              </h5>

src\projects\__tests__\ProjectList-test.js​

...

describe('<ProjectList />', () => {
...

+  test('should display image and remove form when cancel clicked', async () => {
+    setup();

+      screen.getByRole('button', {
+        name: /cancel/i,

+      screen.getByRole('img', {
+        name: /wisozk group/i,

+    ).toBeInTheDocument();
+    expect(
+      screen.queryByRole('form', {

+      })

+  });

});
